    PROCEDURE InitRecord@10();
    BEGIN
      // Start EU.0114065
      IF "General Document" THEN BEGIN
        CASE "Document Type" OF
          "Document Type"::Invoice:
            BEGIN
              IF ("No. Series" <> '') AND
                 (PurchSetup."Gen. Purch. Invoice Nos." = PurchSetup."Posted Gen. Purch. Inv Nos.")
              THEN
                "Posting No. Series" := "No. Series"
              ELSE
                NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Gen. Purch. Inv Nos.");
              IF PurchSetup."Receipt on Purch. Invoice" THEN
                NoSeriesMgt.SetDefaultSeries("Receiving No. Series",PurchSetup."Gen. Purch. Re. Receipt Nos.");
            END;
          "Document Type"::"Credit Memo":
            BEGIN
              IF ("No. Series" <> '') AND
                 (PurchSetup."Gen. Purch. Cr. Memo Nos." = PurchSetup."Posted Gen. Purch. C. Me. Nos.")
              THEN
                "Posting No. Series" := "No. Series"
              ELSE
                NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Gen. Purch. C. Me. Nos.");
                IF PurchSetup."Return Shipment on Credit Memo" THEN
                  NoSeriesMgt.SetDefaultSeries("Return Shipment No. Series",PurchSetup."Gen. Purch. Receipt Nos.");
            END;
        END;
      END ELSE
      // Stop EU.0114065
      // Start INC
          IF "Order Type" = "Order Type"::"Vehicle Sales & Purchases" THEN BEGIN
          CASE "Document Type" OF
            "Document Type"::Inquiry,"Document Type"::Order:
              BEGIN
                NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Vehicle Invoice Nos.");
                NoSeriesMgt.SetDefaultSeries("Receiving No. Series",PurchSetup."Posted Vehicle Receipt Nos.");
              END;
            "Document Type"::Invoice:
              BEGIN
                IF ("No. Series" <> '') AND
                    (PurchSetup."Vehicle Invoice Nos." = PurchSetup."Posted Vehicle Invoice Nos.")
                THEN
                  "Posting No. Series" := "No. Series"
                ELSE
                  NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Vehicle Invoice Nos.");
                IF PurchSetup."Receipt on Purch. Invoice" THEN
                  NoSeriesMgt.SetDefaultSeries("Receiving No. Series",PurchSetup."Posted Vehicle Receipt Nos.");
              END;
            "Document Type"::"Return Order":
              BEGIN
                NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Vehicle Cr. Memo Nos.");
                NoSeriesMgt.SetDefaultSeries("Return Shipment No. Series",PurchSetup."Posted Veh. Return Shpt. Nos.");
              END;
            "Document Type"::"Credit Memo":
              BEGIN
                IF ("No. Series" <> '') AND
                    (PurchSetup."Vehicle Credit Memo Nos." = PurchSetup."Posted Vehicle Cr. Memo Nos.")
                THEN
                  "Posting No. Series" := "No. Series"
                ELSE
                  NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Vehicle Cr. Memo Nos.");
                IF PurchSetup."Return Shipment on Credit Memo" THEN
                  NoSeriesMgt.SetDefaultSeries("Return Shipment No. Series",PurchSetup."Posted Veh. Return Shpt. Nos.");
              END;
            END;
          END ELSE
          CASE "Document Type" OF
            "Document Type"::Inquiry,"Document Type"::Order:
              BEGIN
                NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Invoice Nos.");
                NoSeriesMgt.SetDefaultSeries("Receiving No. Series",PurchSetup."Posted Receipt Nos.");
              END;
            "Document Type"::Invoice:
              BEGIN
                IF ("No. Series" <> '') AND
                    (PurchSetup."Invoice Nos." = PurchSetup."Posted Invoice Nos.")
                THEN
                  "Posting No. Series" := "No. Series"
                ELSE
                  NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Invoice Nos.");
                IF PurchSetup."Receipt on Purch. Invoice" THEN
                  NoSeriesMgt.SetDefaultSeries("Receiving No. Series",PurchSetup."Posted Receipt Nos.");
              END;
            "Document Type"::"Return Order":
              BEGIN
                NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Credit Memo Nos.");
                NoSeriesMgt.SetDefaultSeries("Return Shipment No. Series",PurchSetup."Posted Return Shpt. Nos.");
              END;
            "Document Type"::"Credit Memo":
              BEGIN
                IF ("No. Series" <> '') AND
                    (PurchSetup."Credit Memo Nos." = PurchSetup."Posted Credit Memo Nos.")
                THEN
                  "Posting No. Series" := "No. Series"
                ELSE
                  NoSeriesMgt.SetDefaultSeries("Posting No. Series",PurchSetup."Posted Credit Memo Nos.");
                IF PurchSetup."Return Shipment on Credit Memo" THEN
                  NoSeriesMgt.SetDefaultSeries("Return Shipment No. Series",PurchSetup."Posted Return Shpt. Nos.");
              END;
          END;

      CASE "Document Type" OF
      //  "Document Type"::Order,"Document Type"::Invoice,"Document Type"::"Return Order":
      //    "Order Date" := WORKDATE;
      //  "Document Type"::Invoice:
      //    "Expected Receipt Date" := WORKDATE;
        "Document Type"::Order,"Document Type"::Invoice:
          BEGIN
            "Order Date" := WORKDATE;
            "Expected Receipt Date" := WORKDATE;
          END;
        "Document Type"::"Return Order":
          "Order Date" := WORKDATE;

      END;
      IF ("Document Type" <> "Document Type"::Inquiry) AND ("Posting Date" = 0D) THEN
        "Posting Date" := WORKDATE;
      // Stop INC
      "Document Date" := WORKDATE;

      VALIDATE("Sell-to Customer No.",'');

      IF "Document Type" IN ["Document Type"::"Return Order","Document Type"::"Credit Memo"] THEN BEGIN
        GLSetup.GET;
        Correction := GLSetup."Mark Cr. Memos as Corrections";
      END;

      "Posting Description" := FORMAT("Document Type") + ' ' + "No.";

      // Start INC
      SetStatus(DocStatusLineG.Type::Created); // Set status code for creation
      "Location Code" := UserSessionG.GetLocation;
      "Branch Code" := UserSessionG.GetBranchCode("Location Code");
      IF "Location Code" = '' THEN
        IF CurrFieldNo > 0 THEN
          MESSAGE(C_INC009,USERID);
      IF "Order Type" = "Order Type"::"Vehicle Sales & Purchases" THEN
        "Vehicle Direct Unit Cost Date" := WORKDATE;
      // Stop INC
    END;

          SetStatus(DocStatusLineG.Type::"Ready for Shipment");

          END;
        Status::" ":
          BEGIN
            // If Status is blank, then set to Inserted at least
            SetStatus(DocStatusLineG.Type::Created);

    LOCAL PROCEDURE CalcCurrentStatus@21(ExcludeLineNoP@1001 : Integer);
    VAR
      PurchLineL@1006 : Record 39;
      QuantityL@1005 : Decimal;
      QtyShippedL@1004 : Decimal;
      RetQtyShippedL@1002 : Decimal;
      QtyInvoicedL@1000 : Decimal;
    BEGIN
      QuantityL := 0;
      QtyShippedL := 0;
      RetQtyShippedL := 0;
      QtyInvoicedL := 0;
      CALCFIELDS(Status);
      PurchLineL.RESET;
      PurchLineL.SETRANGE("Document Type","Document Type");
      PurchLineL.SETRANGE("Document No.","No.");
      PurchLineL.SETFILTER("No.",'<>%1','');
      PurchLineL.SETFILTER(Type,'<>%1',PurchLineL.Type::" ");
      PurchLineL.SETFILTER("Line No.", '<>%1', ExcludeLineNoP);
      IF "Document Type" = "Document Type"::"Return Order" THEN
        PurchLineL.SETFILTER("Return Qty. to Ship",'<>0')
      ELSE
        PurchLineL.SETFILTER("Qty. to Receive",'<>0');
      IF PurchLineL.FINDFIRST THEN
        Status := Status::"Ready for Shipment";

      IF "Document Type" = "Document Type"::"Return Order" THEN
        PurchLineL.SETRANGE("Return Qty. to Ship")
      ELSE
        PurchLineL.SETRANGE("Qty. to Receive");
      IF PurchLineL.FINDSET THEN BEGIN
        REPEAT
      // Start EU.0210554.212672
          QuantityL += PurchLineL.Quantity;
          IF "Document Type" = "Document Type"::"Return Order" THEN
            RetQtyShippedL += PurchLineL."Return Qty. Shipped"
          ELSE
            QtyShippedL += PurchLineL."Quantity Received";
      // Stop EU.0210554.212672
          QtyInvoicedL += PurchLineL."Quantity Invoiced";
        UNTIL PurchLineL.NEXT = 0;
        IF ((QuantityL <> 0) AND (QtyShippedL <> 0) AND (QtyShippedL < QuantityL)) OR
           ((QuantityL <> 0) AND (RetQtyShippedL <> 0) AND (RetQtyShippedL < QuantityL))
        THEN
          Status := Status::"Partly Shipped";
      END;

      IF ((QuantityL <> 0) AND (QtyInvoicedL <> 0) AND (QtyInvoicedL < QuantityL)) THEN
        Status := Status::"Partly Invoiced";
      // Start EU.0210554.212672
      IF QuantityL = QtyInvoicedL THEN
        Status := Status::"Completely Invoiced";
      // Stop EU.0210554.212672
    END;
