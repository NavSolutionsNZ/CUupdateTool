        ReportSelections.FilterEmailBodyUsage(Usage);

    PROCEDURE SendEmailInBackground@30(JobQueueEntry@1000 : Record 472);

      ReportUsage@1004 : Integer;
      DocNo@1003 : Code[20];
      DocName@1002 : Text[150];

      EVALUATE(ReportUsage,GetNextParam(ParamString));
      EVALUATE(DocNo,GetNextParam(ParamString));
      EVALUATE(DocName,GetNextParam(ParamString));
      EVALUATE(No,GetNextParam(ParamString));
      // EVALUATE(Type,GetNextParam(ParamString));

        SendEmailToVendorDirectly(ReportUsage,RecRef,DocNo,DocName,FALSE,No,MakeCodeL,LocationCodeL, MainAreaL)
      ELSE
        SendEmailToCustDirectly(ReportUsage,RecRef,DocNo,DocName,FALSE,No,MakeCodeL,LocationCodeL, MainAreaL);
      // Stop EU.0101361
      // Stop EU.0004372
    END;

    LOCAL PROCEDURE GetNextParam@31(VAR Parameter@1000 : Text) : Text;
    VAR
      i@1001 : Integer;
      Result@1002 : Text;
    BEGIN
      i := STRPOS(Parameter,'|');
      IF i > 0 THEN
        Result := COPYSTR(Parameter,1,i - 1);
      IF (i + 1) < STRLEN(Parameter) THEN
        Parameter := COPYSTR(Parameter,i + 1);
      EXIT(Result);
    END;

    PROCEDURE GetReportTitle@1119432000(MakeCodeP@1119432004 : Code[20];LocationCodeP@1119432003 : Code[10];ReportIDP@1119432002 : Integer;ReportNameP@1115000000 : Text[80]) : Text[30];

    BEGIN
      // @ This function returns the Report Title for the report
      // @ The report logic is returned using the following logic
      // @ Prio 1 - Report Title from Report Selection Table
      // @ Prio 2 - Report Name from Report Selection Table
      // @ Prio 3 - Report Name passed from Report if the Report Selection Setup does not exist

      IF ReportSelectionL.FINDFIRST THEN
        IF ReportSelectionL."Print Report Title" THEN
          ReportTitleL := ReportSelectionL."Report Title"
        ELSE BEGIN
          ReportSelectionL.CALCFIELDS("Report Caption");
          ReportTitleL := ReportSelectionL."Report Caption";
        END;

      EXIT(COPYSTR(ReportTitleL,1,30));
      // Stop PA033019
    END;
