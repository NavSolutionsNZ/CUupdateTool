    PROCEDURE DoVendorSelection@1119418003(VAR ResultVendTempP@1119418003 : TEMPORARY Record 23;SearchExpressionP@1109400000 : Code[90]) : Code[90];
    VAR
      AddressL@1119427001 : Record 5029111;
      BlurrySearchExpressionL@1119427005 : Code[90];
    BEGIN
      // Start INC
      // The search algorithm uses field "Name Blurry Key" of the Address table so the search expression has to be turned
      // to a burry search expression. Only address records assigned to vendors will be processed.
      // All found vendors in the target table have a Search Result Ranking made by EvaluateOneFieldMatch from the Address table.

      IF SearchExpressionP > '' THEN
        IF COPYSTR(SearchExpressionP, 1, 1) = '*' THEN
          SearchExpressionP := COPYSTR(SearchExpressionP, 2);
      IF SearchExpressionP > '' THEN
        IF COPYSTR(SearchExpressionP, STRLEN(SearchExpressionP)) = '*' THEN
          SearchExpressionP := COPYSTR(SearchExpressionP, 1, STRLEN(SearchExpressionP) - 1);
      // Start PA042814
      BlurrySearchExpressionL := AddressMgntG.CreateBlurryKeyForSearch(SearchExpressionP,
        MAXSTRLEN(BlurrySearchExpressionL));
      // Stop PA042814

      AddressL.SETCURRENTKEY("Name Blurry Key");
      AddressL.SETFILTER("Name Blurry Key", '*' + BlurrySearchExpressionL + '*');
      // Start EU.0004569
      AddressPrivacyMgtG.AddFilterValidForCompany(AddressL,'');
      // Stop EU.0004569
      IF AddressL.FINDFIRST THEN
        REPEAT
          VendG.RESET;
          VendG.SETCURRENTKEY("Address No.");
          VendG.SETRANGE("Address No.", AddressL."Address No.");
          IF VendG.FINDSET THEN
            REPEAT
              IF NOT ResultVendTempP.GET(VendG."No.") THEN BEGIN
                ResultVendTempP := VendG;
                ResultVendTempP."Search Result Ranking" :=
                  AddressL.EvaluateOneFieldMatch(AddressL."Name Blurry Key", BlurrySearchExpressionL, 3);
                ResultVendTempP."Search Result Ranking" := ResultVendTempP."Search Result Ranking" +
                  AddressL.EvaluateOneFieldMatch(AddressL."Search Name", SearchExpressionP, 6);
                ResultVendTempP.INSERT;
              END;
            UNTIL VendG.NEXT = 0;
        UNTIL AddressL.NEXT = 0;
      VendG.RESET;
      VendG.SETCURRENTKEY("Search Name");
      VendG.SETFILTER("Search Name", '*' + SearchExpressionP + '*');
      VendG.SETFILTER("Address No.", '>0');
      IF VendG.FINDSET THEN
        REPEAT
          IF NOT ResultVendTempP.GET(VendG."No.") THEN BEGIN
            AddressL.GET(VendG."Address No.");
            ResultVendTempP := VendG;
            ResultVendTempP."Search Result Ranking" :=
              AddressL.EvaluateOneFieldMatch(AddressL."Name Blurry Key", BlurrySearchExpressionL, 3);
            ResultVendTempP."Search Result Ranking" := ResultVendTempP."Search Result Ranking" +
              AddressL.EvaluateOneFieldMatch(VendG."Search Name", SearchExpressionP, 6);
            ResultVendTempP.INSERT;
          END;
        UNTIL VendG.NEXT = 0;
      // Stop INC
    END;
