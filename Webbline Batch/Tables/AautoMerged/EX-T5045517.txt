    PROCEDURE FillBuffer@1260700(StatusDateP@1260700 : Date);
    VAR
      InvPostSetupL@1260705 : Record 5813;
      VehicleInventBufferL@1260703 : Record 5045517;
      ValueEntryL@1260702 : Record 5802;
      VehicleCollectUnitL@1260704 : Codeunit 5025409;
      VehicleInventQueryL@1260701 : Query 5045517;
      VehicleInventQuery2L@1260706 : Query 5045517;
      ItemLedgEntryL@1260707 : Record 32;
    BEGIN
      // @ This function is used to read data from 'Value Entry' table and save it in 'Vehicle Inventory Buffer' table
      // @ param StatusDateP - Date

      IF FINDSET THEN
        DELETEALL;
      VehicleInventQueryL.SETFILTER(StatusDate, '<=%1', StatusDateP);
      VehicleInventQueryL.OPEN;
      WHILE VehicleInventQueryL.READ DO BEGIN
        VehicleInventQuery2L.SETRANGE(VIN, VehicleInventQueryL.VIN);
        VehicleInventQuery2L.SETRANGE(Book_No, VehicleInventQueryL.Book_No);
        VehicleInventQuery2L.SETFILTER(StatusDate, '<=%1', StatusDateP);
        VehicleInventQuery2L.SETFILTER(ItemType, '<>%1', ValueEntryL."Item Type"::Item);
        VehicleInventQuery2L.SETFILTER(ExpectedCost, '%1', FALSE);
        VehicleInventQuery2L.SETFILTER(AdjustedCost, '<>%1', 0);
        VehicleInventQuery2L.OPEN;
        IF VehicleInventQuery2L.READ THEN
          InsertLine(VehicleInventQuery2L.VIN, VehicleInventQuery2L.Book_No, StatusDateP)
        ELSE BEGIN
          ItemLedgEntryL.SETCURRENTKEY(VIN,"Book No.");
          ItemLedgEntryL.SETRANGE(VIN, VehicleInventQueryL.VIN);
          ItemLedgEntryL.SETRANGE("Book No.", VehicleInventQueryL.Book_No);
          ItemLedgEntryL.SETRANGE("Item Type", ItemLedgEntryL."Item Type"::Vehicle);
          ItemLedgEntryL.SETFILTER("Posting Date", '<=%1', StatusDateP);
          ItemLedgEntryL.CALCSUMS(Quantity);
          IF ItemLedgEntryL.Quantity <> 0 THEN
            InsertLine(VehicleInventQueryL.VIN, VehicleInventQueryL.Book_No, StatusDateP);
        END;
      END;
    END;

        VehicleInventBufferP."Cylinder Capacity (ccm)" := VehicleL."Cylinder Capacity (ccm)";
        VehicleInventBufferP."Power (kW)" := VehicleL."Power (kW)";

        VehicleInventBufferP."Color Description" := VehicleL."Color Description";

    LOCAL PROCEDURE AssignValuesFromVehiclePriceSticker@1260710(VAR VehicleInventBufferP@1260700 : Record 5045517;StatusDateP@1260702 : Date);

      VehiclePriceStickerHdrL@1260701 : Record 5025432;

    LOCAL PROCEDURE AssignValuesFromVehicleLedgEntry@1260715(VAR VehicleInventBufferP@1260700 : Record 5045517;StatusDateP@1260701 : Date);
    VAR
      VehicleLedgerEntryL@1260702 : Record 5025405;
    BEGIN
      // @ This function is used to copy values from 'Vehicle Ledger Entry' table to VehicleInventBufferP
      // @ param VehicleInventBufferP - Record Vehicle Inventory Buffer
      // @ param StatusDateP - Date

      VehicleLedgerEntryL.RESET;
      VehicleLedgerEntryL.SETCURRENTKEY(VIN,"Entry Type","Not Relevant for Statistics");
      VehicleLedgerEntryL.SETRANGE(VIN, VehicleInventBufferP.VIN);
      VehicleLedgerEntryL.SETRANGE("Entry Type", VehicleLedgerEntryL."Entry Type"::"Service Internal Sale");
      VehicleLedgerEntryL.SETRANGE("Book No.", VehicleInventBufferP."Book No.");
      IF VehicleInventBufferP."Date of Sale" = 0D THEN
        VehicleLedgerEntryL.SETFILTER("Posting Date", '<=%1', StatusDateP);

      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Misc. Portion");
        VehicleInventBufferP."Service Internal" := VehicleLedgerEntryL."Misc. Portion";
      END;

      VehicleLedgerEntryL.SETRANGE("Entry Type", VehicleLedgerEntryL."Entry Type"::Transfer);
      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Branch Fee Amount");
        VehicleInventBufferP."Charging Price Difference" := VehicleLedgerEntryL."Branch Fee Amount";
      END;

      VehicleLedgerEntryL.SETRANGE(Vehicle, TRUE);
      VehicleLedgerEntryL.SETRANGE("Sales Incentive No.", '');
      VehicleLedgerEntryL.SETFILTER("Entry Type", '%1|%2|%3',
         VehicleLedgerEntryL."Entry Type"::Purchase,
         VehicleLedgerEntryL."Entry Type"::"Pos. Adjmt.",
         VehicleLedgerEntryL."Entry Type"::Transfer);
      VehicleLedgerEntryL.SETFILTER("Line Type", '%1|%2', VehicleLedgerEntryL."Line Type"::Item, VehicleLedgerEntryL."Line Type"::Vehicle);

      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Vehicle Portion", "Manuf. Option Portion");
        VehicleInventBufferP.Vehicle := VehicleLedgerEntryL."Vehicle Portion";
        VehicleInventBufferP."Manufacturer Options" := VehicleLedgerEntryL."Manuf. Option Portion";
      END;

      VehicleLedgerEntryL.SETRANGE("Line Type", VehicleLedgerEntryL."Line Type"::Item);
      VehicleLedgerEntryL.SETFILTER("Sales Incentive No.", '<>%1', '');
      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Vehicle Portion");
        VehicleInventBufferP."Sales Incentive (Received)" := VehicleLedgerEntryL."Vehicle Portion";
      END;

      VehicleLedgerEntryL.SETRANGE("Sales Incentive No.");
      VehicleLedgerEntryL.SETRANGE(Vehicle, FALSE);
      VehicleLedgerEntryL.SETRANGE("Line Type", VehicleLedgerEntryL."Line Type"::"Own Option");
      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Own Option Portion");
        VehicleInventBufferP."Own Options" := VehicleLedgerEntryL."Own Option Portion";
      END;

      VehicleLedgerEntryL.SETRANGE("Line Type", VehicleLedgerEntryL."Line Type"::"Veh. Transfer");
      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Veh. Transfer Portion");
        VehicleInventBufferP."Veh. Trasfer" := VehicleLedgerEntryL."Veh. Transfer Portion";
      END;

      VehicleLedgerEntryL.SETFILTER("Entry Type", '%1|%2', VehicleLedgerEntryL."Entry Type"::"Other Costs", VehicleLedgerEntryL."Entry Type"::"Other Income");
      VehicleLedgerEntryL.SETRANGE("Line Type", VehicleLedgerEntryL."Line Type"::"G/L Account");
      VehicleLedgerEntryL.SETRANGE(Vehicle, FALSE);
      VehicleLedgerEntryL.SETFILTER("Sales Incentive No.", '<>%1', '');
      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Misc. Portion");
        VehicleInventBufferP."Sales Incentives (Expected)" := VehicleLedgerEntryL."Misc. Portion";
      END;

      VehicleLedgerEntryL.SETFILTER("Entry Type", '%1|%2',
         VehicleLedgerEntryL."Entry Type"::Purchase,
         VehicleLedgerEntryL."Entry Type"::Transfer);
      VehicleLedgerEntryL.SETRANGE("Sales Incentive No.", '');
      IF VehicleLedgerEntryL.FINDSET THEN BEGIN
        VehicleLedgerEntryL.CALCSUMS("Misc. Portion");
        VehicleInventBufferP.Miscellaneous := VehicleLedgerEntryL."Misc. Portion";
      END;

      VehicleInventBufferP."Purchase Total" :=
         VehicleInventBufferP.Vehicle +
         VehicleInventBufferP."Manufacturer Options" +
         VehicleInventBufferP."Own Options" +
         VehicleInventBufferP."Veh. Trasfer" +
         VehicleInventBufferP."Sales Incentive (Received)" +
         VehicleInventBufferP."Sales Incentives (Expected)" +
         VehicleInventBufferP."Service Internal" +
         VehicleInventBufferP.Miscellaneous +
         VehicleInventBufferP."Calculated Costs";

      IF VehicleInventBufferP."Sales Total" <> 0 THEN
        VehicleInventBufferP."Gross Profit" := VehicleInventBufferP."Sales Total" - VehicleInventBufferP."Purchase Total";
    END;

    LOCAL PROCEDURE AssignCalcCostFromVehicleLedgEntry@1260702(VAR VehicleInventBufferP@1260701 : Record 5045517;StatusDateP@1260700 : Date);
    VAR
      VehicleLedgerEntryL@1260702 : Record 5025405;
    BEGIN
      // @ This function is used to copy values for Calculated Costs from 'Vehicle Ledger Entry' table to VehicleInventBufferP
      // @ param VehicleInventBufferP - Record Vehicle Inventory Buffer
      // @ param StatusDateP - Date

      VehicleLedgerEntryL.RESET;
      VehicleLedgerEntryL.SETCURRENTKEY("Entry Type",VIN,"Book No.","Posting Date");
      VehicleLedgerEntryL.SETRANGE("Entry Type", VehicleLedgerEntryL."Entry Type"::"Calcd. Costs");
      VehicleLedgerEntryL.SETRANGE("Book No.", VehicleInventBufferP."Book No.");
      VehicleLedgerEntryL.SETRANGE(VIN, VehicleInventBufferP.VIN);
      IF VehicleInventBufferP."Date of Sale" = 0D THEN
        VehicleLedgerEntryL.SETFILTER("Posting Date", '<=%1', StatusDateP);
      VehicleLedgerEntryL.CALCSUMS("Calculated Costs Portion");
      VehicleInventBufferP."Calculated Costs" := VehicleLedgerEntryL."Calculated Costs Portion";
    END;

    LOCAL PROCEDURE AssignSalesData@1260701(VAR VehicleInventBufferP@1260700 : Record 5045517;StatusDateP@1260702 : Date);
    VAR
      VehicleLedgerEntryL@1260701 : Record 5025405;
    BEGIN
      // @ This function is used to copy sales data from 'Vehicle Ledger Entry' table to VehicleInventBufferP
      // @ param VehicleInventBufferP - Record Vehicle Inventory Buffer
      // @ param StatusDateP - Date

      VehicleLedgerEntryL.RESET;
      VehicleLedgerEntryL.SETCURRENTKEY("Chain Book No.","Book No.","Entry Type",VIN,"Date of Sale");
      VehicleLedgerEntryL.SETRANGE("Book No.", VehicleInventBufferP."Book No.");
      VehicleLedgerEntryL.SETRANGE(VIN, VehicleInventBufferP.VIN);
      VehicleLedgerEntryL.SETRANGE("Line Type", VehicleLedgerEntryL."Line Type"::Vehicle);
      VehicleLedgerEntryL.SETRANGE(Vehicle, TRUE);
      VehicleLedgerEntryL.CALCSUMS(Quantity);
      IF VehicleLedgerEntryL.Quantity = 0 THEN BEGIN
        VehicleLedgerEntryL.SETRANGE("Entry Type", VehicleLedgerEntryL."Entry Type"::Sale);
        VehicleLedgerEntryL.SETFILTER("Line Type",'%1|%2|%3|%4',
           VehicleLedgerEntryL."Line Type"::Vehicle,
           VehicleLedgerEntryL."Line Type"::"Own Option",
           VehicleLedgerEntryL."Line Type"::"Veh. Transfer",
           VehicleLedgerEntryL."Line Type"::"G/L Account");
        VehicleLedgerEntryL.SETRANGE(Vehicle);
        VehicleLedgerEntryL.CALCSUMS("Total Price");
        VehicleInventBufferP."Sales Total" := ABS(VehicleLedgerEntryL."Total Price");
        IF VehicleLedgerEntryL.FINDLAST THEN BEGIN
          VehicleInventBufferP.Customer := VehicleLedgerEntryL."Source No.";
          VehicleInventBufferP."Date of Sale" := VehicleLedgerEntryL."Date of Sale";
          VehicleInventBufferP."Sales Document No." := VehicleLedgerEntryL."Document No.";
      // START EU.0106426.108439
          IF VehicleInventBufferP.Customer <> '' THEN
            VehicleInventBufferP."Vehicle Status" := VehicleInventBufferP."Vehicle Status"::"Customer Vehicle";
      // START EU.0106426.108439
        END;
      END;
    END;

    LOCAL PROCEDURE CheckAdjustmentNeeded@1260704(VAR VehicleInventBufferP@1260701 : Record 5045517;StatusDateP@1260700 : Date);
    VAR
      ValueEntryL@1260702 : Record 5802;
    BEGIN
      IF VehicleInventBufferP."Sales Total" <> 0 THEN BEGIN
        ValueEntryL.RESET;
        ValueEntryL.SETCURRENTKEY("Item No.","Posting Date","Location Code","Book No.","Document No.","Item Ledger Entry Type");
        ValueEntryL.SETRANGE(VIN, VehicleInventBufferP.VIN);
        ValueEntryL.SETRANGE("Book No.", VehicleInventBufferP."Book No.");
        IF VehicleInventBufferP."Date of Sale" = 0D THEN
          ValueEntryL.SETFILTER("Posting Date", '<=%1', StatusDateP);
        ValueEntryL.SETFILTER("Item Type", '<>%1', ValueEntryL."Item Type"::Item);
        ValueEntryL.SETRANGE("Expected Cost", FALSE);
        ValueEntryL.CALCSUMS("Adjusted Cost");
        VehicleInventBufferP.Adjust := ValueEntryL."Adjusted Cost" <> 0;
      END;
    END;

    LOCAL PROCEDURE InsertLine@1260703(VINP@1260700 : Code[17];BookNoP@1260701 : Code[20];StatusDateP@1260704 : Date);
    VAR
      ValueEntryL@1260702 : Record 5802;
      VehicleInventBufferL@1260703 : Record 5045517;
      InvPostSetupL@1260705 : Record 5813;
    BEGIN
      // @ This function is used to insert a new line in 'Vehicle Inventory Buffer' table
      // @ param VINP - Code 17
      // @ param BookNoP - Code 20
      // @ param StatusDateP - Date

      IF VehicleInventBufferL.GET(VINP, BookNoP, USERID) THEN
        EXIT;
      ValueEntryL.RESET;
      // Start EU.0158119.159359
      ValueEntryL.SETCURRENTKEY(VIN,"Posting Date","Book No.","Item Type");
      // Stop EU.0158119.159359
      ValueEntryL.SETRANGE(VIN, VINP);
      ValueEntryL.SETRANGE("Book No.", BookNoP);
      ValueEntryL.SETFILTER("Posting Date", '<=%1', StatusDateP);
      ValueEntryL.SETFILTER("Item Type", '<>%1', ValueEntryL."Item Type"::Item);
      IF ValueEntryL.FINDSET THEN BEGIN
        VehicleInventBufferL.INIT;
        VehicleInventBufferL.VIN := VINP;
        VehicleInventBufferL."Book No." := BookNoP;
        VehicleInventBufferL."User ID" := USERID;
        ValueEntryL.SETRANGE("Expected Cost", FALSE);
        ValueEntryL.CALCSUMS("Cost Posted to G/L");
        VehicleInventBufferL."G/L Account Value" := ValueEntryL."Cost Posted to G/L";

        ValueEntryL.CALCSUMS("Adjusted Cost");
        VehicleInventBufferL."Inventory Value" := ValueEntryL."Adjusted Cost";

        ValueEntryL.SETRANGE("Item Type", ValueEntryL."Item Type"::Vehicle);
        ValueEntryL.SETRANGE("Expected Cost");
        VehicleInventBufferL."Purchase Date" := ValueEntryL."Posting Date";

        ValueEntryL.FINDLAST;
        VehicleInventBufferL."Vehicle Status" := ValueEntryL."Vehicle Status";
        VehicleInventBufferL."Location Code" := ValueEntryL."Location Code";
        VehicleInventBufferL."Branch Code" := ValueEntryL."Branch Code";
        VehicleInventBufferL."Inv. Posting Group" := ValueEntryL."Inventory Posting Group";
        InvPostSetupL.RESET;
        IF InvPostSetupL.GET(VehicleInventBufferL."Location Code", VehicleInventBufferL."Inv. Posting Group") THEN
          VehicleInventBufferL."Inventory Account" := InvPostSetupL."Inventory Account";

        AssignSalesData(VehicleInventBufferL, StatusDateP);
        IF StatusDateP < VehicleInventBufferL."Date of Sale" THEN
          VehicleInventBufferL."Days Carried" := VehicleInventBufferL."Date of Sale" - VehicleInventBufferL."Purchase Date"
        ELSE
          VehicleInventBufferL."Days Carried" := StatusDateP - VehicleInventBufferL."Purchase Date";

        AssignValuesFromVehicleCard(VehicleInventBufferL);
        AssignValuesFromVehiclePriceSticker(VehicleInventBufferL, StatusDateP);
        AssignValuesFromVehicleLedgEntry(VehicleInventBufferL, StatusDateP);
        AssignCalcCostFromVehicleLedgEntry(VehicleInventBufferL, StatusDateP);
        CheckAdjustmentNeeded(VehicleInventBufferL, StatusDateP);
        VehicleInventBufferL.INSERT;
      END;
    END;
