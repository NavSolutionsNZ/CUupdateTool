    PROCEDURE FillBuffer@1260700(StatusDateP@1260700 : Date);
    VAR
      InvPostSetupL@1260705 : Record 5813;
      VehicleInventBufferL@1260703 : Record 5045517;
      ValueEntryL@1260702 : Record 5802;
      VehicleCollectUnitL@1260704 : Codeunit 5025409;
      VehicleInventQueryL@1260701 : Query 5045517;
      VehicleInventQuery2L@1260706 : Query 5045517;
      ItemLedgEntryL@1260707 : Record 32;
    BEGIN
      // @ This function is used to read data from 'Value Entry' table and save it in 'Vehicle Inventory Buffer' table
      // @ param StatusDateP - Date

      IF FINDSET THEN
        DELETEALL;
      VehicleInventQueryL.SETFILTER(StatusDate, '<=%1', StatusDateP);
      VehicleInventQueryL.OPEN;
      WHILE VehicleInventQueryL.READ DO BEGIN
        VehicleInventQuery2L.SETRANGE(VIN, VehicleInventQueryL.VIN);
        VehicleInventQuery2L.SETRANGE(Book_No, VehicleInventQueryL.Book_No);
        VehicleInventQuery2L.SETFILTER(StatusDate, '<=%1', StatusDateP);
        VehicleInventQuery2L.SETFILTER(ItemType, '<>%1', ValueEntryL."Item Type"::Item);
        VehicleInventQuery2L.SETFILTER(ExpectedCost, '%1', FALSE);
        VehicleInventQuery2L.SETFILTER(AdjustedCost, '<>%1', 0);
        VehicleInventQuery2L.OPEN;
        IF VehicleInventQuery2L.READ THEN
          InsertLine(VehicleInventQuery2L.VIN, VehicleInventQuery2L.Book_No, StatusDateP)
        ELSE BEGIN
          ItemLedgEntryL.SETCURRENTKEY(VIN,"Book No.");
          ItemLedgEntryL.SETRANGE(VIN, VehicleInventQueryL.VIN);
          ItemLedgEntryL.SETRANGE("Book No.", VehicleInventQueryL.Book_No);
          ItemLedgEntryL.SETRANGE("Item Type", ItemLedgEntryL."Item Type"::Vehicle);
          ItemLedgEntryL.SETFILTER("Posting Date", '<=%1', StatusDateP);
          ItemLedgEntryL.CALCSUMS(Quantity);
          IF ItemLedgEntryL.Quantity <> 0 THEN
            InsertLine(VehicleInventQueryL.VIN, VehicleInventQueryL.Book_No, StatusDateP);
        END;
      END;
    END;

    LOCAL PROCEDURE AssignSalesData@1260701(VAR VehicleInventBufferP@1260700 : Record 5045517;StatusDateP@1260702 : Date);
    VAR
      VehicleLedgerEntryL@1260701 : Record 5025405;
    BEGIN
      // @ This function is used to copy sales data from 'Vehicle Ledger Entry' table to VehicleInventBufferP
      // @ param VehicleInventBufferP - Record Vehicle Inventory Buffer
      // @ param StatusDateP - Date

      VehicleLedgerEntryL.RESET;
      // Start EU.0166467.166817
      // Stop EU.0166467.166817
      VehicleLedgerEntryL.SETRANGE("Book No.", VehicleInventBufferP."Book No.");
      VehicleLedgerEntryL.SETRANGE(VIN, VehicleInventBufferP.VIN);
      // Start EU.0177226.177066
      VehicleLedgerEntryL.SETRANGE("Posting Date", 0D, StatusDateP);
      // Stop EU.0177226.177066
      VehicleLedgerEntryL.SETRANGE("Line Type", VehicleLedgerEntryL."Line Type"::Vehicle);
      VehicleLedgerEntryL.SETRANGE(Vehicle, TRUE);
      VehicleLedgerEntryL.CALCSUMS(Quantity);

        VehicleLedgerEntryL.SETRANGE("Entry Type", VehicleLedgerEntryL."Entry Type"::Sale);

           VehicleLedgerEntryL."Line Type"::"Own Option",
           VehicleLedgerEntryL."Line Type"::"Veh. Transfer",
           VehicleLedgerEntryL."Line Type"::"G/L Account");

        VehicleLedgerEntryL.CALCSUMS("Total Price");
        VehicleInventBufferP."Sales Total" := ABS(VehicleLedgerEntryL."Total Price");

          VehicleInventBufferP.Customer := VehicleLedgerEntryL."Source No.";

          VehicleInventBufferP."Sales Document No." := VehicleLedgerEntryL."Document No.";
      // Start EU.0106426.108439
      // Start EU.0172844.172952
      // Start EU.0188843.191149
         SetVehicleStatus(VehicleInventBufferP,VehicleLedgerEntryL."Vehicle Status");
      // Stop EU.0188843.191149
      // Stop EU.0172844.172952
      // Stop EU.0106426.108439
        END;
      END;
    END;

    LOCAL PROCEDURE InsertLine@1260703(VINP@1260700 : Code[17];BookNoP@1260701 : Code[20];StatusDateP@1260704 : Date);
    VAR
      ValueEntryL@1260702 : Record 5802;
      VehicleInventBufferL@1260703 : Record 5045517;
      InvPostSetupL@1260705 : Record 5813;
    BEGIN
      // @ This function is used to insert a new line in 'Vehicle Inventory Buffer' table
      // @ param VINP - Code 17
      // @ param BookNoP - Code 20
      // @ param StatusDateP - Date

      IF VehicleInventBufferL.GET(VINP, BookNoP, USERID) THEN
        EXIT;
      ValueEntryL.RESET;
      // Start EU.0166467.166817
      // Start EU.0158119.159359
      ValueEntryL.SETCURRENTKEY(VIN,"Book No.","Posting Date","Item Type");
      // Stop EU.0158119.159359
      ValueEntryL.SETRANGE(VIN, VINP);
      ValueEntryL.SETRANGE("Book No.", BookNoP);
      ValueEntryL.SETFILTER("Posting Date", '<=%1', StatusDateP);
      ValueEntryL.SETFILTER("Item Type", '<>%1', ValueEntryL."Item Type"::Item);
      IF ValueEntryL.FINDFIRST THEN BEGIN
        VehicleInventBufferL.INIT;
        VehicleInventBufferL.VIN := VINP;
        VehicleInventBufferL."Book No." := BookNoP;
        VehicleInventBufferL."User ID" := USERID;
        ValueEntryL.SETRANGE("Expected Cost", FALSE);
        ValueEntryL.CALCSUMS("Cost Posted to G/L");
        VehicleInventBufferL."G/L Account Value" := ValueEntryL."Cost Posted to G/L";

        ValueEntryL.CALCSUMS("Adjusted Cost");
        VehicleInventBufferL."Inventory Value" := ValueEntryL."Adjusted Cost";

        ValueEntryL.SETRANGE("Item Type", ValueEntryL."Item Type"::Vehicle);
        ValueEntryL.SETRANGE("Expected Cost");
        VehicleInventBufferL."Purchase Date" := ValueEntryL."Posting Date";

        ValueEntryL.FINDLAST;
        VehicleInventBufferL."Vehicle Status" := ValueEntryL."Vehicle Status";
        VehicleInventBufferL."Location Code" := ValueEntryL."Location Code";
        VehicleInventBufferL."Branch Code" := ValueEntryL."Branch Code";
        VehicleInventBufferL."Inv. Posting Group" := ValueEntryL."Inventory Posting Group";
        InvPostSetupL.RESET;
        IF InvPostSetupL.GET(VehicleInventBufferL."Location Code", VehicleInventBufferL."Inv. Posting Group") THEN
          VehicleInventBufferL."Inventory Account" := InvPostSetupL."Inventory Account";

        AssignSalesData(VehicleInventBufferL, StatusDateP);
      // Start EU.0177226.177066
        IF ValueEntryL."Invoiced Quantity" = 0 THEN
          VehicleInventBufferL."Sales Total" := 0;
      // Stop EU.0177226.177066
        IF StatusDateP < VehicleInventBufferL."Date of Sale" THEN
          VehicleInventBufferL."Days Carried" := VehicleInventBufferL."Date of Sale" - VehicleInventBufferL."Purchase Date"
        ELSE
          VehicleInventBufferL."Days Carried" := StatusDateP - VehicleInventBufferL."Purchase Date";

        AssignValuesFromVehicleCard(VehicleInventBufferL);
        AssignValuesFromVehiclePriceSticker(VehicleInventBufferL, StatusDateP);
      // Start EU.0177226.177066
