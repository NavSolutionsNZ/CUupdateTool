    PROCEDURE FilterPostedInvoices@17452300(VAR RecP@17452300 : Record 112;OrderNoP@17452301 : Code[20];OrderTypeP@17452302 : Option);
    VAR
      LineL@17452303 : Record 113;
    BEGIN
      // @ This function will filter the Sales Invoices related to the provided "Order No." and "Order Type" checking the link in the document line level.
      // Start EU.0174523
      // Prepayment Invoices are being checked in header level
      RecP.RESET;
      RecP.SETRANGE("Prepayment Order No.", OrderNoP);
      RecP.SETRANGE(Prepayment, TRUE);
      IF RecP.FINDSET THEN
        REPEAT
          RecP.MARK(TRUE);
        UNTIL RecP.NEXT = 0;
      RecP.SETRANGE(Prepayment);
      RecP.SETRANGE("Prepayment Order No.");
      // Regular Invoices are being checked in line level
      LineL.RESET;
      LineL.FILTERGROUP(-1);
      LineL.SETRANGE("Service Order No.", OrderNoP);
      LineL.SETRANGE("Order No.", OrderNoP);
      LineL.FILTERGROUP(0);
      LineL.SETRANGE("Order Type", OrderTypeP);
      IF LineL.FINDSET THEN
        REPEAT
          RecP.MARK(RecP.GET(LineL."Document No."));
          LineL.SETFILTER("Document No.", '>%1', LineL."Document No.");
        UNTIL LineL.NEXT = 0;
      RecP.SETCURRENTKEY("No.");
      RecP.MARKEDONLY(TRUE);
      // Stop EU.0174523
    END;
