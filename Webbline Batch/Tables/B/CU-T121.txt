    PROCEDURE NoOnLookup@5025351(VAR Text@1109400000 : Text[250]) Change : Boolean;
    VAR
      DataL@1109400001 : Record 5600;
      ItemL@1109400002 : Record 27;
      StandardTextsL@1109400003 : Record 7;
      GLAccL@1109400004 : Record 15;
      ServMakeL@1109400006 : Record 5028685;
      ItemChargeL@1109400007 : Record 5800;
    BEGIN
      // Start INC
      CASE Type OF
        Type::Item:
          BEGIN
            ItemL.SETCURRENTKEY("Item Type","Make Code","No.");
            IF "Order Type" = "Order Type"::Service THEN BEGIN
              ServMakeL.GetServiceMakeFilter("Shortcut Dimension 2 Code");
              ServMakeL.COPYFILTER("Service Make Code", ItemL."Make Code");
            END ELSE
              IF "Shortcut Dimension 2 Code" <> '' THEN
                ItemL.SETFILTER("Make Code",'%1|%2',"Shortcut Dimension 2 Code",'');

            CASE "Item Type" OF
              "Item Type"::Vehicle:
                BEGIN
                  IF ItemL.GET(Text) THEN
                    ;
                  IF PAGE.RUNMODAL(PAGE::"Model List", ItemL) = ACTION::LookupOK THEN BEGIN
                    Change := TRUE;
                    Text := ItemL."No.";
                  END;
                END;

              "Item Type"::"Veh. Transfer":
                BEGIN
                  IF ItemL.GET(Text) THEN
                    ;
                  IF PAGE.RUNMODAL(PAGE::"Transfer Fees List", ItemL) = ACTION::LookupOK THEN BEGIN
                    Text := ItemL."No.";
                    Change := TRUE;
                  END;
                END;

              "Item Type"::Item:
                BEGIN
                  IF ItemL.GET(Text) THEN
                    ;
                  IF PAGE.RUNMODAL(0, ItemL) = ACTION::LookupOK THEN BEGIN
                    Text := ItemL."No.";
                    Change := TRUE;
                  END;
                END;
            END;
          END;
        Type::"Account (G/L)":
          BEGIN
            GLAccL.SETRANGE("Direct Posting",TRUE);
            GLAccL.SETRANGE(Blocked,FALSE);
            IF GLAccL.GET(Text) THEN
              ;
            IF PAGE.RUNMODAL(0, GLAccL) = ACTION::LookupOK THEN BEGIN
              Text := GLAccL."No.";
              Change := TRUE;
            END;
          END;
        Type::" ":
          BEGIN
            IF StandardTextsL.GET(Text) THEN
              ;
            IF PAGE.RUNMODAL(0, StandardTextsL) = ACTION::LookupOK THEN BEGIN
              Text := StandardTextsL.Code;
              Change := TRUE;
            END;
          END;
        Type::"Fixed Asset":
          BEGIN
            IF DataL.GET("No.") THEN
              ;
            IF PAGE.RUNMODAL(0, DataL) = ACTION::LookupOK THEN BEGIN
              Text := DataL."No.";
              Change := TRUE;
            END;
          END;
        Type::"Charge (Item)":
          BEGIN
            IF ItemChargeL.GET(Text) THEN
              ;
            IF PAGE.RUNMODAL(0, ItemChargeL) = ACTION::LookupOK THEN BEGIN
              Text := ItemChargeL."No.";
              Change := TRUE;
            END;
          END;
      END;
      // Stop INC
    END;
