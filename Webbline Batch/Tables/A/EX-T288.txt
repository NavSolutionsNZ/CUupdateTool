    PROCEDURE CheckLetterSalutation@5025353();
    VAR
      ContactNewL@1109400000 : Text[200];
    BEGIN
      // Start PA036268
      IF "Letter Salutation Code" = '' THEN
        ContactNewL := DELCHR(Name,'<>')
      ELSE
        IF CurrFieldNo <> 0 THEN
          ContactNewL := AddressSetupG.CreateContactValue(
            (CurrFieldNo = 0),
            "Address Salutation Code",
            "Letter Salutation Code",
            "Title Code",
            Name,
            "First Name",
            "Last Name",
            "Name 2",
            Contact);
      IF ContactNewL <> Contact THEN
        VALIDATE(Contact, COPYSTR(ContactNewL, 1, MAXSTRLEN(Contact)));
      // Stop PA036268
    END;
