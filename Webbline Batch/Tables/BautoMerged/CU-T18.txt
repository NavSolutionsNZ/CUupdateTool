    PROCEDURE DoCustomerSelection@1119418003(VAR ResultCustTempP@1119418003 : TEMPORARY Record 18;SearchExpressionP@1109400000 : Code[90]) : Code[90];
    VAR
      AddressL@1119427001 : Record 5029111;
      BlurrySearchExpressionL@1119427005 : Code[90];
    BEGIN
      // Start INC
      // The search algorithm uses field "Name Blurry Key" of the Address table so the search expression has to be turned
      // to a burry search expression. Only address records assigned to customers will be processed.
      // All found customers in the target table have a Search Result Ranking made by EvaluateOneFieldMatch from the Address table.

      IF SearchExpressionP > '' THEN
        IF COPYSTR(SearchExpressionP, 1, 1) = '*' THEN
          SearchExpressionP := COPYSTR(SearchExpressionP, 2);
      IF SearchExpressionP > '' THEN
        IF COPYSTR(SearchExpressionP, STRLEN(SearchExpressionP)) = '*' THEN
          SearchExpressionP := COPYSTR(SearchExpressionP, 1, STRLEN(SearchExpressionP) - 1);
      // Start PA042814
      BlurrySearchExpressionL := AddressMgtG.CreateBlurryKeyForSearch(SearchExpressionP, MAXSTRLEN(BlurrySearchExpressionL));
      // Stop PA042814

      AddressL.SETCURRENTKEY("Name Blurry Key");
      AddressL.SETFILTER("Name Blurry Key", '*' + BlurrySearchExpressionL + '*');
      // Start EU.0004569
      AddressPrivacyMgtG.AddFilterValidForCompany(AddressL,'');
      // Stop EU.0004569
      IF AddressL.FINDSET THEN
        REPEAT
          CustG.RESET;
          CustG.SETCURRENTKEY("Address No.");
          CustG.SETRANGE("Address No.", AddressL."Address No.");
          CustG.SETFILTER("Branch Posting Group", GETFILTER("Branch Posting Group"));
          IF CustG.FINDSET THEN
            REPEAT
              IF NOT ResultCustTempP.GET(CustG."No.") THEN BEGIN
                ResultCustTempP := CustG;
                ResultCustTempP."Search Result Ranking" :=
                  AddressL.EvaluateOneFieldMatch(AddressL."Name Blurry Key", BlurrySearchExpressionL, 3);
                ResultCustTempP."Search Result Ranking" := ResultCustTempP."Search Result Ranking" +
                  AddressL.EvaluateOneFieldMatch(AddressL."Search Name", SearchExpressionP, 6);
                ResultCustTempP.INSERT;
              END;
            UNTIL CustG.NEXT = 0;
        UNTIL AddressL.NEXT = 0;

      CustG.RESET;
      CustG.SETCURRENTKEY("Search Name");
      CustG.SETFILTER("Search Name", '*' + SearchExpressionP + '*');
      CustG.SETFILTER("Branch Posting Group", GETFILTER("Branch Posting Group"));
      CustG.SETFILTER("Address No.", '>0');
      IF CustG.FINDSET THEN
        REPEAT
          IF NOT ResultCustTempP.GET(CustG."No.") THEN BEGIN
            AddressL.GET(CustG."Address No.");
            ResultCustTempP := CustG;
            ResultCustTempP."Search Result Ranking" :=
              AddressL.EvaluateOneFieldMatch(AddressL."Name Blurry Key", BlurrySearchExpressionL, 3);
            ResultCustTempP."Search Result Ranking" := ResultCustTempP."Search Result Ranking" +
              AddressL.EvaluateOneFieldMatch(CustG."Search Name", SearchExpressionP, 6);
            ResultCustTempP.INSERT;
          END;
        UNTIL CustG.NEXT = 0;
      // Stop INC
    END;
