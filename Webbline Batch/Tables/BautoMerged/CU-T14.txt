    PROCEDURE PurchaseCrossLocationCheck@1101224021(LineTypeP@1101224021 : Text[30];HdrLocationP@1101224020 : Code[10];LineLocationP@1101224019 : Code[10];OrderTypeP@1109400000 : Integer);
    VAR
      PurchHeaderL@1109400002 : Record 38;
      PurchLineLocationSyncL@1109400001 : Boolean;
      PurchLineL@1109400003 : Record 39;
    BEGIN
      // @ Ths function will check, if the entered location in a purchase order line matched the
      // @ location comming from the purchase order header. In case it isn't then an error is shown depending
      // @ on the PurchasesPayablesSetupG."Purchase Line Location Sync" parameter
      // Start PA036118
      IF LineTypeP = ' ' THEN // --> Test if Line Type = Text
        EXIT;
      IF HdrLocationP = LineLocationP THEN
        EXIT;
      // Start PA036549.34230
      CASE OrderTypeP OF
        PurchHeaderL."Order Type"::"Parts Sales & Purchases" :
          BEGIN
            InventorySetupG.GET;
            PurchLineLocationSyncL := InventorySetupG."Purch. Line Location Sync.";
          END;
        PurchHeaderL."Order Type"::"Vehicle Sales & Purchases" :
          BEGIN
            VehicleSetupG.GET;
            PurchLineLocationSyncL := VehicleSetupG."Purch. Line Location Sync.";
          END;
      END;
      // Start PA042464.82666
      // Start PA041187.83979
      IF NOT PurchLineLocationSyncL THEN
        EXIT;
      IF OrderTypeP = PurchHeaderL."Order Type"::"Parts Sales & Purchases" THEN
      // Start PA041187.83976
        IF LineTypeP = FORMAT(PurchLineL.Type::"Account (G/L)") THEN
          EXIT;
      // Stop PA041187.83976
      // Stop PA042464.82666
      // Stop PA036549.34230
      ERROR(C_INC_LineHdrLocationMustMatch, ServDeptG.FIELDCAPTION("Location Code"));
      // Stop PA041187.83979
      // Stop PA036118
    END;

    PROCEDURE SalesCrossLocationCheck@1101224019(LineTypeP@1101224021 : Text[30];HdrLocationP@1101224020 : Code[10];LineLocationP@1101224019 : Code[10];OrderTypeP@1109400000 : Integer;DocumentTypeP@1055600 : Integer);
    VAR
      SalesHeaderL@1109400001 : Record 36;
      SalesLineL@1109400003 : Record 37;
      SalesLineLocationSyncL@1109400002 : Boolean;
    BEGIN
      // @ Ths function will check, if the entered location in a sales order line matched the
      // @ location comming from the sales order header. In case it isn't then an error is shown depending
      // @ on the SalesReceivablesSetupG."Sales Line Location Sync" parameter
      // Start PA036118
      IF LineTypeP = ' ' THEN // --> Test if Line Type = Text
        EXIT;
      IF HdrLocationP = LineLocationP THEN
        EXIT;
      // Start PA036549.34230
      CASE OrderTypeP OF
        SalesHeaderL."Order Type"::"Parts Sales & Purchases" :
          BEGIN
            InventorySetupG.GET;
            SalesLineLocationSyncL := InventorySetupG."Sales Line Location Sync.";
          END;
        SalesHeaderL."Order Type"::"Vehicle Sales & Purchases" :

            VehicleSetupG.GET;
            SalesLineLocationSyncL := VehicleSetupG."Sales Line Location Sync.";
          END;
        SalesHeaderL."Order Type"::Service :
          BEGIN
            ServiceSetupG.GET();
            SalesLineLocationSyncL := ServiceSetupG."Service Line Location Sync";
          END;
      END;
      // Start PA041187.87194
      IF NOT SalesLineLocationSyncL THEN
        EXIT;
      // Start PA042464.83967
      IF OrderTypeP = SalesHeaderL."Order Type"::"Parts Sales & Purchases" THEN
        IF DocumentTypeP IN [SalesHeaderL."Document Type"::Invoice, SalesHeaderL."Document Type"::"Credit Memo"] THEN
          IF LineTypeP = FORMAT(SalesLineL.Type::"Account (G/L)") THEN
            EXIT;

        IF DocumentTypeP IN [SalesHeaderL."Document Type"::Quote, SalesHeaderL."Document Type"::Order] THEN

    PROCEDURE SalesCrossBranchCheck@1101224025(SalesHdrP@1101224022 : Record 36;SalesLineP@1101224021 : Record 37);
    VAR
      CrossBranchL@1101368291 : Boolean;
      UserLocationExistsL@1101368292 : Boolean;
      SalesLineL@1109400000 : Record 37;
    BEGIN
      // @ This function will check, if the location in the entered sales line blongs to the same branch as the location of the header or
      // @ not.
      // Start PA036118
      IF SalesLineP.Type = SalesLineP.Type::" " THEN
        EXIT;
      IF SalesLineP."Location Code" = '' THEN
        EXIT;
      IF SalesLineP."Location Code" = SalesHdrP."Location Code" THEN
        EXIT;
      IF SalesHdrP."Order Type" = SalesHdrP."Order Type"::"Vehicle Sales & Purchases" THEN BEGIN
      // Start PA036512.30330
        CrossBranchL := DoCrossBranchCheck(SalesHdrP."Location Code", SalesLineP."Location Code",
          TRUE,SalesHdrP."Document Type" = SalesHdrP."Document Type"::Quote);
        IF ShowConfirmationG AND
      // Start PA041187.87194
           (UserLocationExist(SalesHdrP."Location Code", SalesLineP."Location Code"))
        THEN BEGIN
          IF (NOT CrossBranchL) OR (CrossBranchL AND ( SalesHdrP."Document Type" = SalesHdrP."Document Type"::Quote)) THEN BEGIN
      // Start EU.0058995
            IF GUIALLOWED THEN
              IF NOT CONFIRM(STRSUBSTNO(C_INC_VinLocDIfferentConfirm,SalesLineP.VIN),FALSE) THEN
                ERROR('');
      // Stop EU.0058995
          END ELSE
            ERROR(C_INC_LocationDifferentError);
        END ELSE
          IF ( SalesHdrP."Document Type" = SalesHdrP."Document Type"::Quote) OR (SalesHdrP."Document Type" = SalesHdrP."Document Type"::Order) THEN BEGIN
            UserLocationExistsL := UserLocationExist(SalesHdrP."Location Code", SalesLineP."Location Code");
            CASE TRUE OF
              UserLocationExistsL AND CrossBranchL:
                BEGIN
                  UserSessionUnitG.ShowMessageBox(STRSUBSTNO(C_INC_BranchDifferentMsg, SalesLineP.VIN));
                  EXIT;
                END;
              (NOT UserLocationExistsL) AND CrossBranchL:
                BEGIN
                  UserSessionUnitG.ShowMessageBox(STRSUBSTNO(C_INC_BranchDifferentMsg, SalesLineP.VIN));
                  EXIT;
                END;
            END;
          END ELSE BEGIN
            IF CrossBranchL THEN
              ERROR(C_INC_LocationDifferentError);
            IF NOT ShowConfirmationG THEN
              UserSessionUnitG.ShowMessageBox(STRSUBSTNO(C_INC_VehLocationDifferentMsg, SalesLineP.VIN));
          END;
      // Start PA036512.30531
      // Stop PA036512.30531
      END ELSE BEGIN
        IF SalesHdrP."Order Type" = SalesLineP."Order Type"::"Parts Sales & Purchases" THEN BEGIN
          IF NOT((SalesLineP.Type = SalesLineP.Type::"Account (G/L)") AND
             (SalesLineP."Document Type" IN [SalesLineP."Document Type"::Invoice,SalesLineP."Document Type"::"Credit Memo"]))
          THEN
            DoCrossBranchCheck(SalesHdrP."Location Code", SalesLineP."Location Code", FALSE,FALSE);
        END ELSE
      // Start PA042464.83967
          DoCrossBranchCheck(SalesHdrP."Location Code", SalesLineP."Location Code", FALSE,FALSE);
      // Stop PA042464.83967
      // Stop PA041187.87194
      // Stop PA036512.30330
        IF SalesLineP.Type = SalesLineP.Type::Item THEN
          UserSessionUnitG.ShowMessageBox(STRSUBSTNO(C_INC_LocationDifferentMsg, SalesLineP."Item Type", SalesLineP."No."))
        ELSE
          UserSessionUnitG.ShowMessageBox(STRSUBSTNO(C_INC_LocationDifferentMsg, SalesLineP.Type, SalesLineP."No."));
      END;
      // Stop PA036118
    END;
