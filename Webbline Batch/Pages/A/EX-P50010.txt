    PROCEDURE TestBatch@2();
    VAR
      GenJnlLine3@1000000002 : Record 81;
      GenJnlLine4@1000000000 : Record 81;
      VendorTotal@1000000003 : Decimal;
      CurrencyCode@1000000005 : Code[10];
      Lastvendor@1000000004 : Code[20];
      BankAccNo@1000000001 : Code[20];
      LastBankAccNo@1000000006 : Code[20];
      BankL@1101353000 : Record 270;
      VendorL@1101353001 : Record 23;
      VendorBankL@1101353002 : Record 288;
    BEGIN
      // Start DC5.00
      // Check the Bank Account
      GenJnlLine3.RESET;
      GenJnlLine3.SETCURRENTKEY("Account Type","Account No.","Document Type","Document No.");
      GenJnlLine3.SETRANGE("Journal Template Name", "Journal Template Name");
      GenJnlLine3.SETRANGE("Journal Batch Name", "Journal Batch Name");
      GenJnlLine3.SETRANGE("Account Type", "Account Type"::"Bank Account");

      IF NOT GenJnlLine3.FIND('-') THEN BEGIN
        // Check that all lines have a Bal. Account No. and they are all the same.
        GenJnlLine4.RESET;
        GenJnlLine4.SETRANGE("Journal Template Name", "Journal Template Name");
        GenJnlLine4.SETRANGE("Journal Batch Name", "Journal Batch Name");
        GenJnlLine4.FIND('-');
        LastBankAccNo := GenJnlLine4."Bal. Account No.";
        BankAccNo := GenJnlLine4."Bal. Account No.";
        REPEAT
          IF GenJnlLine4."Bal. Account No." <> BankAccNo THEN
            GenJnlLine4.FIELDERROR(
              GenJnlLine4."Bal. Account No.",
              'must be the same on all lines');
          IF GenJnlLine4."Bal. Account Type" <> GenJnlLine4."Bal. Account Type"::"Bank Account" THEN
            GenJnlLine4.FIELDERROR(
              "Bal. Account Type",
              'must be "Bank Account"');
          IF GenJnlLine4."Bal. Account No." = '' THEN
            GenJnlLine4.FIELDERROR(
              GenJnlLine4."Bal. Account No.",
              'cannot be blank');
        UNTIL GenJnlLine4.NEXT = 0;
      END ELSE BEGIN
        IF GenJnlLine3.COUNT = 1 THEN BEGIN
          // Check that no lines have a Bal. Account No.
          GenJnlLine3.FIND('-');
          BankAccNo := GenJnlLine3."Account No.";
          GenJnlLine4.RESET;
          GenJnlLine4.SETRANGE("Journal Template Name", "Journal Template Name");
          GenJnlLine4.SETRANGE("Journal Batch Name", "Journal Batch Name");
          GenJnlLine4.FIND('-');
          REPEAT
            IF GenJnlLine4."Bal. Account No." <> '' THEN
              GenJnlLine4.FIELDERROR(
                GenJnlLine4."Bal. Account No.",
                'must be blank');
          UNTIL GenJnlLine4.NEXT = 0;
        END;

        IF GenJnlLine3.COUNT > 1 THEN
          ERROR('There must not be more than one Bank Account Journal line.');
      END;

      GenJnlLine3.RESET;
      GenJnlLine3.SETRANGE("Journal Template Name", "Journal Template Name");
      GenJnlLine3.SETRANGE("Journal Batch Name", "Journal Batch Name");
      GenJnlLine3.FIND('-');
      IF GenJnlLine3."Currency Code" <> CurrencyCode THEN
        ERROR('All Journal lines must have the same %1.', GenJnlLine3.FIELDCAPTION(GenJnlLine3."Currency Code"));

      // Check the Bank
      BankL.GET(BankAccNo);
      BankL.TESTFIELD("Bank Code");
      CASE BankL."Bank Code" OF
        BankL."Bank Code"::ANZ,BankL."Bank Code"::BNZ,BankL."Bank Code"::NAT,BankL."Bank Code"::ASB:
          BEGIN
            BankL.TESTFIELD("Transfer File Folder");
            BankL.TESTFIELD("Transfer File Name Code");
            BankL.TESTFIELD("DC Bank No.");
            BankL.TESTFIELD("DC Bank Branch No.");
            BankL.TESTFIELD("DC Bank Account No.");
            BankL.TESTFIELD("Subscriber Name");
            BankL.VALIDATE("Transfer File Folder");
          END;
        BankL."Bank Code"::DeskBank:
          BEGIN
            BankL.TESTFIELD("Transfer File Folder");
            BankL.TESTFIELD("Transfer File Name Code");
            BankL.TESTFIELD("DC Bank No.");
            BankL.TESTFIELD("DC Bank Branch No.");
            BankL.TESTFIELD("DC Bank Account No.");
            BankL.TESTFIELD("Subscriber Name");
            BankL.VALIDATE("Transfer File Folder");
          END;
      END;

      // Check the Vendors.
      GenJnlLine3.RESET;
      GenJnlLine3.SETCURRENTKEY("Account Type","Account No.","Document Type","Document No.");
      GenJnlLine3.SETRANGE("Journal Template Name", "Journal Template Name");
      GenJnlLine3.SETRANGE("Journal Batch Name", "Journal Batch Name");
      GenJnlLine3.SETRANGE("Account Type", "Account Type"::Vendor);
      GenJnlLine3.FIND('-');
      CurrencyCode := GenJnlLine3."Currency Code";
      Lastvendor := GenJnlLine3."Account No.";
      CLEAR(VendorTotal);
      REPEAT
        IF GenJnlLine3."Currency Code" <> CurrencyCode THEN
          ERROR('All Journal lines must have the same %1.', GenJnlLine3.FIELDCAPTION(GenJnlLine3."Currency Code"));
        VendorL.GET(GenJnlLine3."Account No.");
        VendorL.TESTFIELD("Direct Cr. Bank Account");
        VendorBankL.GET(GenJnlLine3."Account No.", VendorL."Direct Cr. Bank Account");
        CASE BankL."Bank Code" OF
          BankL."Bank Code"::ANZ,BankL."Bank Code"::BNZ,BankL."Bank Code"::NAT,BankL."Bank Code"::ASB:
            BEGIN
              VendorBankL.TESTFIELD("DC Bank No.");
              VendorBankL.TESTFIELD("DC Bank Branch No.");
              VendorBankL.TESTFIELD("DC Bank Account No.");
              VendorBankL.TESTFIELD("Other Party Name");
            END;
          BankL."Bank Code"::DeskBank:
            BEGIN
              VendorBankL.TESTFIELD("DC Bank No.");
              VendorBankL.TESTFIELD("DC Bank Branch No.");
              VendorBankL.TESTFIELD("DC Bank Account No.");
              VendorBankL.TESTFIELD("DC Bank Suffix");
              VendorBankL.TESTFIELD("Other Party Name");
            END;
        END;
        IF Lastvendor <> GenJnlLine3."Account No." THEN BEGIN
          IF NOT (VendorTotal > 0) THEN
            ERROR('Total Amount for Vendor %1 must be greater than zero.', GenJnlLine3."Account No.");
          CLEAR(VendorTotal);
          Lastvendor := GenJnlLine3."Account No.";
        END;
        VendorTotal := VendorTotal + GenJnlLine3.Amount;
      UNTIL GenJnlLine3.NEXT = 0;

      IF NOT (VendorTotal > 0) THEN
        ERROR('Total Amount for Vendor %1 must be greater than zero.', GenJnlLine3."Account No.");

      // Stop DC5.00
    END;
