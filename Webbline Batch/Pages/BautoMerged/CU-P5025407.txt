    PROCEDURE CheckPricingControl@1109400006();
    VAR
      UserSetupL@1109400000 : Record 91;
    BEGIN
      // @ Checks the fields in the 'Pricing Control' tab in the User Setup. If the user has insufficient rights some fields are set to
      // @ 'EDITABLE' = FALSE;

      "Unit PriceEditable" := TRUE;
      "Unit Price incl. VATEditable" := TRUE;
      "List Price (LCY)Editable" := TRUE;

        ListPriceinclVATLCYEditable := FALSE;

    END;
