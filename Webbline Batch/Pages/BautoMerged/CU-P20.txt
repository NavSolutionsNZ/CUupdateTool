    PROCEDURE UnApplyReconciliation@48401();
    VAR
      GLEntriesL@48402 : Record 17;
      ReconcilationEntriesL@48401 : Record 5204813;
      ReconcilationEntriesTempL@48400 : TEMPORARY Record 5204813;
      NoOfRecordsL@572200 : Integer;
    BEGIN
      // @ Function to Unapply selected G/L Entries

      // Start EU.0000484
      CurrPage.SETSELECTIONFILTER(GLEntriesL);

      ReconcilationEntriesL.RESET;
      ReconcilationEntriesL.SETCURRENTKEY("G/L Entry No.");

      // Start EU.0005722
      NoOfRecordsL := GLEntriesL.COUNT();
      // Stop EU.0005722

      GLEntriesL.FINDSET(TRUE,FALSE);
      REPEAT
        ReconcilationEntriesL.SETRANGE("G/L Entry No.",GLEntriesL."Entry No.");
        ReconcilationEntriesL.SETRANGE("Entry Type",ReconcilationEntriesL."Entry Type"::Initial);
        IF ReconcilationEntriesL.FINDFIRST() THEN BEGIN
      // Start EU.0005722
          IF ReconcilationEntriesL.Amount = ReconcilationEntriesL."Amount to Reconcile" THEN
            IF NoOfRecordsL > 1 THEN
              ERROR(C_INC_UnApplyErr,GLEntriesL."Entry No.")
            ELSE
              ERROR(C_INC_UnApplyErr2);
      // Stop EU.0005722
          ReconcilationEntriesTempL := ReconcilationEntriesL;
          ReconcilationEntriesTempL.INSERT;
      // Start EU.0005722
        END ELSE
          IF NoOfRecordsL > 1 THEN
            ERROR(C_INC_UnApplyErr,GLEntriesL."Entry No.")
          ELSE
            ERROR(C_INC_UnApplyErr2);
      // Stop EU.0005722
      UNTIL  GLEntriesL.NEXT=0;

      ReconcilationEntriesL.UnApplyApplication(ReconcilationEntriesTempL);
      // Stop EU.0000484
    END;
