    PROCEDURE CalculateTempItemUsage@2(ViewCompressedP@1109400000 : Boolean);
    BEGIN
      TempItemUsageG.RESET;

      TempItemUsageG.DELETEALL;
      UsageQty12MG := 0;
      NoOfLines12MG := 0;
      // Start PA032509
      //"Manufacturer Req. Status" := '';
      // Stop PA032509
      "Replacement Status" := "Replacement Status"::" ";

      IF (Type <> Type::Item) AND ("No." = '') THEN
        EXIT;

      IF NOT ItemG.GET("No.") THEN
        EXIT;

      TempItemUsageG.INIT;
      TempItemUsageG."Item No." := "No.";

      IF ViewCompressedP THEN
        TempItemUsageG."Calculation level" := TempItemUsageG."Calculation level"::Item
      ELSE BEGIN
        TempItemUsageG."Calculation level" := TempItemUsageG."Calculation level"::Bin;
        TempItemUsageG."Location Code" := "Location Code";
        TempItemUsageG."Bin Code" := "Bin Code";
      END;
      CalculateItemUsageG.CalcRequi("No.",
        TempItemUsageG,
        CALCDATE('<-12M>',WORKDATE),
        WORKDATE);
      UsageQty12MG := TempItemUsageG."Sale Quantity";
      NoOfLines12MG := TempItemUsageG."Line Count";

      // Start PA032509
      CALCFIELDS("Replacement Status");
      // Stop PA032509
    END;

    LOCAL PROCEDURE ShowTransferFromLocationCodes@37000() : Text;
    VAR
      ItemTransferSuggestionL@37001 : Record 5028738;
      LocationCodeStringL@37000 : Text;

      // @ This function will show for the current Rec all active "From Loaction codes + transfer qty in a string.
      // Start EU.0000370
      ItemTransferSuggestionL.SETRANGE("Worksheet Template Name", "Worksheet Template Name");
      ItemTransferSuggestionL.SETRANGE("Journal Batch Name", "Journal Batch Name");
      ItemTransferSuggestionL.SETRANGE("Line No.", "Line No.");

      IF ItemTransferSuggestionL.FINDSET THEN BEGIN
        REPEAT
          LocationCodeStringL := LocationCodeStringL + '; ' + ItemTransferSuggestionL."From Location Code";
          IF ItemTransferSuggestionL."From Bin Code" > '' THEN
            LocationCodeStringL := LocationCodeStringL + ',' + ItemTransferSuggestionL."From Bin Code";
          LocationCodeStringL := LocationCodeStringL + ' (' + FORMAT(ItemTransferSuggestionL."Quantity to Transfer") + ')';
        UNTIL ItemTransferSuggestionL.NEXT = 0;
        LocationCodeStringL := COPYSTR(LocationCodeStringL, 3);

    END;
